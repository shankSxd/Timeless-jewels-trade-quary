1.Skopiuj poniższy kod i wklej do konsoli otwieranej za pomocą zbadaj element na stronie timeless jewel generator:

function yoink() {
    let y = [];
    arr = Array.from(document.getElementsByClassName('font-bold text-orange-500 text-center'));
    arr.forEach(element => {
        y.push(element.innerHTML.replace(/Seed |\(weight [0-9]+\)/g, '').trim())
    });
    return y;
}
